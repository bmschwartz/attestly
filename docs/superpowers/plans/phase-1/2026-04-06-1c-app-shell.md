# Sub-Plan 1c: App Shell Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create the app shell with an auth-aware navbar, all route stubs, and client-side route protection that redirects unauthenticated users to login for protected pages.

**Architecture:** A shared `Navbar` component renders in the root layout and adapts its content based on auth state (authenticated, unauthenticated, survey-response mode). All routes are created as empty page stubs with the correct file-system routing structure. Protected routes use a client-side `AuthGuard` wrapper that checks Privy auth state and triggers the login modal if the user is not authenticated.

**Tech Stack:** Next.js 16 App Router, Privy React Auth (`usePrivy`), Tailwind CSS 4, React 19

**Spec reference:** `docs/superpowers/specs/2026-04-05-respondent-experience-design.md`

### Task 1: Create the AuthGuard component

**Files:**
- Create: `src/app/_components/auth-guard.tsx`

- [ ] **Step 1: Create the AuthGuard wrapper**

Create `src/app/_components/auth-guard.tsx`:

```tsx
"use client";

import { usePrivy } from "@privy-io/react-auth";
import { useEffect } from "react";

/**
 * AuthGuard — wraps protected pages.
 *
 * If the user is not authenticated, triggers the Privy login modal.
 * Shows a loading state while auth is being determined.
 * Renders children only when the user is authenticated.
 */
export function AuthGuard({ children }: { children: React.ReactNode }) {
  const { ready, authenticated, login } = usePrivy();

  useEffect(() => {
    if (ready && !authenticated) {
      login();
    }
  }, [ready, authenticated, login]);

  if (!ready) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <div className="text-gray-500">Loading...</div>
      </div>
    );
  }

  if (!authenticated) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <div className="text-gray-500">Redirecting to login...</div>
      </div>
    );
  }

  return <>{children}</>;
}
```

### Task 2: Create the Navbar component

**Files:**

- [ ] **Step 1: Create the auth-aware navbar**

Create `src/app/_components/navbar.tsx`:

```tsx
"use client";

import Link from "next/link";
import { usePathname } from "next/navigation";
import { usePrivy } from "@privy-io/react-auth";
import { useState, useRef, useEffect } from "react";

/**
 * Navbar — adapts based on auth state and current route.
 *
 * Authenticated: Logo | Explore | Dashboard | My Responses | [Avatar dropdown with sign out]
 * Unauthenticated: Logo | Explore | [Sign In]
 * During survey response (/s/[slug]/respond): minimal (logo + save status)
 */
export function Navbar() {
  const pathname = usePathname();
  const { ready, authenticated, login, logout, user } = usePrivy();
  const [dropdownOpen, setDropdownOpen] = useState(false);
  const dropdownRef = useRef<HTMLDivElement>(null);

  // Close dropdown when clicking outside
  useEffect(() => {
    function handleClickOutside(event: MouseEvent) {
      if (
        dropdownRef.current &&
        !dropdownRef.current.contains(event.target as Node)
      ) {
        setDropdownOpen(false);
      }
    }
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  // Minimal navbar during survey response
  const isSurveyResponse = /^\/s\/[^/]+\/respond$/.test(pathname);
  if (isSurveyResponse) {
    return (
      <nav className="border-b border-gray-200 bg-white px-4 py-3">
        <div className="mx-auto flex max-w-7xl items-center justify-between">
          <Link href="/" className="text-xl font-bold text-indigo-600">
            Attestly
          </Link>
          <span className="text-sm text-gray-400">
            {/* Save status will be injected by the response page */}
          </span>
        </div>
      </nav>
    );
  }

  const navLinkClasses = (href: string) => {
    const isActive = pathname === href;
    return `text-sm font-medium transition-colors ${
      isActive
        ? "text-indigo-600"
        : "text-gray-600 hover:text-gray-900"
    }`;
  };

  return (
    <nav className="border-b border-gray-200 bg-white px-4 py-3">
      <div className="mx-auto flex max-w-7xl items-center justify-between">
        {/* Left: Logo + nav links */}
        <div className="flex items-center gap-6">
          <Link href="/" className="text-xl font-bold text-indigo-600">
            Attestly
          </Link>
          <Link href="/explore" className={navLinkClasses("/explore")}>
            Explore
          </Link>
          {ready && authenticated && (
            <>
              <Link
                href="/dashboard"
                className={navLinkClasses("/dashboard")}
              >
                Dashboard
              </Link>
              <Link
                href="/my-responses"
                className={navLinkClasses("/my-responses")}
              >
                My Responses
              </Link>
            </>
          )}
        </div>

        {/* Right: Auth actions */}
        <div className="flex items-center">
          {!ready ? (
            <div className="h-8 w-8 animate-pulse rounded-full bg-gray-200" />
          ) : authenticated ? (
            <div className="relative" ref={dropdownRef}>
              <button
                onClick={() => setDropdownOpen(!dropdownOpen)}
                className="flex h-8 w-8 items-center justify-center rounded-full bg-indigo-100 text-sm font-medium text-indigo-700 transition-colors hover:bg-indigo-200"
                aria-label="User menu"
              >
                {getUserInitial(user)}
              </button>
              {dropdownOpen && (
                <div className="absolute right-0 mt-2 w-48 rounded-md border border-gray-200 bg-white py-1 shadow-lg">
                  <div className="border-b border-gray-100 px-4 py-2">
                    <p className="truncate text-sm font-medium text-gray-900">
                      {getUserDisplayName(user)}
                    </p>
                    <p className="truncate text-xs text-gray-500">
                      {getUserEmail(user)}
                    </p>
                  </div>
                  <Link
                    href="/settings/profile"
                    className="block px-4 py-2 text-sm text-gray-700 hover:bg-gray-50"
                    onClick={() => setDropdownOpen(false)}
                  >
                    Settings
                  </Link>
                  <button
                    onClick={() => {
                      setDropdownOpen(false);
                      void logout();
                    }}
                    className="block w-full px-4 py-2 text-left text-sm text-gray-700 hover:bg-gray-50"
                  >
                    Sign out
                  </button>
                </div>
              )}
            </div>
          ) : (
            <button
              onClick={login}
              className="rounded-md bg-indigo-600 px-4 py-2 text-sm font-medium text-white transition-colors hover:bg-indigo-700"
            >
              Sign In
            </button>
          )}
        </div>
      </div>
    </nav>
  );
}

/**
 * Extract the first initial from the Privy user for the avatar.
 */
function getUserInitial(user: ReturnType<typeof usePrivy>["user"]): string {
  if (!user) return "?";
  const name =
    user.google?.name ??
    user.apple?.email ??
    user.email?.address ??
    "";
  return name.charAt(0).toUpperCase() || "?";
}

/**
 * Extract a display name from the Privy user.
 */
function getUserDisplayName(
  user: ReturnType<typeof usePrivy>["user"],
): string {
  if (!user) return "User";
  return (
    user.google?.name ??
    user.email?.address ??
    user.apple?.email ??
    "User"
  );
}

/**
 * Extract an email from the Privy user.
 */
function getUserEmail(
  user: ReturnType<typeof usePrivy>["user"],
): string {
  if (!user) return "";
  return (
    user.email?.address ??
    user.google?.email ??
    user.apple?.email ??
    ""
  );
}
```
